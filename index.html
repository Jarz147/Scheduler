// Constants const MAINTENANCE_INTERVAL_DAYS = 30; // Change this for different intervals (e.g., 90 for quarterly) // Load machines from localStorage let machines = JSON.parse(localStorage.getItem('machines')) || []; // DOM elements const form = document.getElementById('machine-form'); const machineNameInput = document.getElementById('machine-name'); const lastMaintenanceInput = document.getElementById('last-maintenance'); const machineList = document.getElementById('machine-list'); // Function to calculate next maintenance date function calculateNextDate(lastDate) { const date = new Date(lastDate); date.setDate(date.getDate() + MAINTENANCE_INTERVAL_DAYS); return date.toISOString().split('T')[0]; // Format as YYYY-MM-DD } // Function to render the machine table function renderMachines() { machineList.innerHTML = ''; machines.forEach((machine, index) => { const row = document.createElement('tr'); row.innerHTML = ` ${machine.name} ${machine.lastMaintenance} ${machine.nextMaintenance} `; machineList.appendChild(row); }); } // Function to add or update a machine form.addEventListener('submit', (e) => { e.preventDefault(); const name = machineNameInput.value.trim(); const lastMaintenance = lastMaintenanceInput.value; if (name && lastMaintenance) { const nextMaintenance = calculateNextDate(lastMaintenance); // Check if machine already exists (update) or add new const existingIndex = machines.findIndex(m => m.name.toLowerCase() === name.toLowerCase()); if (existingIndex >= 0) { machines[existingIndex] = { name, lastMaintenance, nextMaintenance }; } else { machines.push({ name, lastMaintenance, nextMaintenance }); } // Save to localStorage and re-render localStorage.setItem('machines', JSON.stringify(machines)); renderMachines(); // Clear form machineNameInput.value = ''; lastMaintenanceInput.value = ''; } }); // Function to delete a machine function deleteMachine(index) { machines.splice(index, 1); localStorage.setItem('machines', JSON.stringify(machines)); renderMachines(); } // Initial render renderMachines();
